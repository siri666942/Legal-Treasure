backend/
├── app/                    # 主应用目录（类似 Java 的 src/main/java）
│   ├── __init__.py         # Python 包标识文件
│   ├── main.py             # 应用入口（类似 Spring Boot 的 Application.java）
│   │
│   ├── api/                # API 层（类似 Controller 层）
│   │   ├── router.py       # 路由聚合
│   │   ├── deps.py         # 依赖注入（类似 @Autowired）
│   │   └── endpoints/      # 具体接口（类似各个 Controller）
│   │
│   ├── core/               # 核心配置（类似 application.properties）
│   │   └── config.py       # 配置类
│   │
│   ├── db/                 # 数据库层（类似 DAO/Repository）
│   │   ├── base.py         # ORM 基类
│   │   └── session.py      # 数据库会话（类似 EntityManager）
│   │
│   ├── models/             # 数据库表映射
│   ├── schemas/            # 数据传输对象（类似 DTO）
│   └── services/           # 业务逻辑层（类似 Service 层）
